			imageLoaderCounter--;

			// all images loaded
			if (imageLoaderCounter == 0) {
				imageContainers.each(function() {
					var container = $(this);
					var img = container.find("img");
					var src = img.attr("src");
					img.css("visibility", "hidden");
					container.css({
						"display": "block",
						"background-image": "url('" + src + "')",
						"background-attachment": "fixed",
						"background-position": "center 0px",
						"background-size": "cover"
					});
					if( isHandheld() ) {
						container.css({
							"background-attachment": "initial",
							"background-position": "center center"
						});
					}
				});

				adjustImageDimensions();
			}
		}
		
		/**
		 * On mobile, remove any CSS for the parallax effect
		 */
		function imageLoadedMobile() {
			imageContainers.each(function() {
				var container = $(this);
				container.attr("style", "");
				container.find("img").css("visibility", "visible");
			});
		}

		/**
		 * Load larger images 
		 */
		function imageLoader(mode) {
			imageContainers.each(function() {
				var container = $(this);
				var newImage = container.find("img").attr("data-"+mode+"-src");
				var img = document.createElement('img');
				if(mode == 'desktop') {
					img.onload = imageLoadedDesktop;
					imageLoaderCounter++;
				} else {
					img.onload = imageLoadedMobile;
				}
				img.src = newImage;
				container.find("span.imageWrapper img").prop('src', newImage);
			});
		}

		/**
		 * Scrolling - animate the background image
		 */
		function handleScrolling() {
			if( isHandheld() ) return;
			
			var win = $(this);
			var scrollTop = win.scrollTop();
			var viewportHeight = win.height();

			imageContainers.each(function() {
				var container = $(this);
				var containerHeight = container.height();
				var imageHeight = container.find("img").height();
				var top = container.position().top;
				var min = top - viewportHeight;
				var max = top + containerHeight;
				var scrollingFactor = imageHeight - containerHeight;
				if (scrollTop >= min && scrollTop <= max) {
					perc = 0.5 - (scrollTop - min) / (max - min);
					container.css("background-position", "center " + Math.round(scrollingFactor * perc) + "px");
				}
			});
		}
		
		function reinit_overlap() {
			$(window).off('scroll', handleScrolling);
			if (!isMobile()) {
				// load larger images on desktop
				imageLoader('desktop');

				// animate the background container background a little bit
				handleScrolling();
				$(window).on('scroll', handleScrolling);
			} else {
				imageLoader('mobile');
			}
		}
		
		/* Re-initiate gallery with a throttled resize event. */
		var reinitTimer;
		$(window).resize(function() {
			clearTimeout(reinitTimer);
			reinitTimer = setTimeout(reinit_overlap, 100);
		});
		reinit_overlap();

	});

</script>
